function startDownload() { const url = document.getElementById('url').value; const q = document.getElementById('quality').value; const f = document.getElementById('format').value; const e = document.getElementById('ext').value; if(!url) return alert("Enter a URL first!"); const btn = document.getElementById('dl-btn'); const status = document.getElementById('status'); const statusText = document.getElementById('status-text'); // UI Transformation btn.disabled = true; btn.innerText = "🔍 DETECTING REAL NAME..."; status.classList.remove('hidden'); // Redirect to download window.location.href = `/download?url=${encodeURIComponent(url)}&quality=${q}&format=${f}&ext=${e}`; // Update status messages every few seconds setTimeout(() => { btn.innerText = "⚡ MERGING AUDIO/VIDEO..."; }, 3000); setTimeout(() => { btn.innerText = "🚀 START DOWNLOAD"; btn.disabled = false; statusText.innerText = "✅ SENT TO BROWSER!"; statusText.classList.replace('text-red-500', 'text-green-500'); }, 15000); }
